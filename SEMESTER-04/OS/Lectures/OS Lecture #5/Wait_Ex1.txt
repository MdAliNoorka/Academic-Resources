#include <iostream>
#include <unistd.h>
#include <sys/wait.h>
using namespace std;
int main()
{
//int n,a,b,status;
pid_t ret_value,PID;
cout<<"Process Id is: "<<getpid()<<endl;
ret_value=fork();
if(ret_value<0)
{
//fork has failed #
cout<<"fork failure..."<<endl;
}else if(ret_value==0)
{
//Child Process
cout<<"Child Process:"<<endl;
cout<<"Child Process Id is: "<<getpid()<<" and parent process Id is: "<<getppid()<<endl;
//sleep(20);
}
else
{
int status;
PID=wait(&status);
cout<<"Status: "<<status<<endl;
cout<<"Terminated Process Id: "<<PID<<endl;

//Parent Process
cout<<"Parent Process:"<<endl;
cout<<"Process Id is: "<<getpid()<<endl;

//sleep(30);
}
return 0;
}
